### 直接模拟

```C++ [ ]
class Solution {
public:
    vector<int> selfDividingNumbers(int left, int right) {
        // 创建结果数组
        vector<int> list;
        // 如果有满足要求的就将其放入数组中
        for (int i = left; i <= right; i++)
            if (check(i)) list.push_back(i);
        // 返回列表
        return list;
    }

    bool check(int i) {
        int t = i;
        // 依次取出 数字 i 中的各个数字，与数字 i 进行比较
        while (t != 0) {
            int q = t % 10;
            // 如果数字中包含 0 或者 无法整除，直接返回 false
            if (q == 0) return false;
            if (i % q != 0) return false;
            t /= 10;
        }
        // 如果都成功了，就返回 true
        return true;
    }
};
```
